dpkg(1)
installation (dpkg -i)
	extract ctl files
	run pre-inst script
	unpack new - backup old
	postrm old
	configure:
		unpack new conffiles - backup old (will restore on error)
		run postinst script
removal
	run prerm script
	remove installed files
	run postrm script
